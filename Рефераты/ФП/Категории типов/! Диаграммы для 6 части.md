\usepackage{tikz}
\usepackage{tikz-cd}
\usepackage{xcolor}
\usepackage{amsmath}
\usetikzlibrary{backgrounds, fit}
\usetikzlibrary{decorations.pathmorphing}

\tikzcdset{scale cd/.style=
  {every
    label/.append style={scale=#1},
    cells={nodes={scale=#1}},
    arrows={line width=#1, -{Stealth[scale=1]}},
  },
}
\begin{document}
\boldmath
\begin{tikzcd}[scale cd=2, sep=10em]

\mathcal{A}
\ar[rd, ""{name=KD}, "K/D"', double distance=2]
\ar[rd, ""{name=F}, "F", bend left=60, double distance=2]
\ar["u", from=F, to=KD, squiggly, Red, shorten >= -5mm, shorten <= 1mm]
&
\\
\mathcal{I}
\ar[u, "K", double distance=2]
\ar[r, "D"', double distance=2]
&
\mathcal{C}

\end{tikzcd}
\end{document}
