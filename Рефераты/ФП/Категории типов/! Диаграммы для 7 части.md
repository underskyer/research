
# Объект категории запятой
```tikz
\usepackage{tikz}
\usepackage{tikz-cd}
\usepackage{xcolor}
\usetikzlibrary{backgrounds, fit}
\usetikzlibrary{decorations.pathmorphing}
\usetikzlibrary{shapes.geometric}
\tikzcdset{scale cd/.style=
  {every
    label/.append style={scale=#1, transform shape},
    cells={nodes={scale=#1, transform shape}},
    arrows={ultra thick, -{Stealth[scale=1]}},
  },
}
\begin{document}

\boldmath
%\mathversion{bold}

\begin{tikzcd}[
  scale cd=2,
  sep=10em,
  column sep=20em,
  /tikz/execute at end picture={
	\begin{scope}[on background layer]
		\node[
		  draw=black!80,       % Обводка (как в filldraw)
		  fill=Blue!10,        % Заливка (как в filldraw)
		  very thick,          % Толщина линии
		  circle,              % Форма
		  minimum size=3.6cm,  % Диаметр (2 * 1.8 из вашего примера)
		  fit=(a),             % Охватываем узел (a)
		  label={[font=\huge]140:$\mathcal{A}$}      % Подпись СНАРУЖИ сверху
		] (ca) {};

		\node[
		  draw=black!80,       % Обводка (как в filldraw)
		  fill=Green!10,        % Заливка (как в filldraw)
		  very thick,          % Толщина линии
		  circle,              % Форма
		  minimum size=3.6cm,  % Диаметр (2 * 1.8 из вашего примера)
		  fit=(b),             % Охватываем узел (a)
		  label={[font=\huge]140:$\mathcal{B}$}      % Подпись СНАРУЖИ сверху
		] (cb) {};

		\node[
		  draw=black!80,       % Обводка (как в filldraw)
		  fill=Red!10,        % Заливка (как в filldraw)
		  very thick,          % Толщина линии
		  ellipse,              % Форма
		  minimum height=10cm,
		  minimum width=10cm,
		  fit=(Fa) (Gb),
		  label={[font=\huge]140:$\mathcal{C}$}      % Подпись СНАРУЖИ сверху
		] (cFaGb) {};

		\node[
		  draw=black!80,       % Обводка (как в filldraw)
		  fill=Blue!10,        % Заливка (как в filldraw)
		  very thick,          % Толщина линии
		  circle,              % Форма
		  minimum size=3.6cm,  % Диаметр (2 * 1.8 из вашего примера)
		  fit=(Fa),             % Охватываем узел (a)
		  label={[font=\huge]30:$F\mathcal{A}$}      % Подпись СНАРУЖИ сверху
		] (cFa) {};

		\node[
		  draw=black!80,       % Обводка (как в filldraw)
		  fill=Green!10,        % Заливка (как в filldraw)
		  very thick,          % Толщина линии
		  circle,              % Форма
		  minimum size=3.6cm,  % Диаметр (2 * 1.8 из вашего примера)
		  fit=(Gb),             % Охватываем узел (a)
		  label={[font=\huge]30:$G\mathcal{B}$}      % Подпись СНАРУЖИ сверху
		] (cGb) {};
	\end{scope}
  }
]

|[alias=a]| a
\ar[r, "F"{pos=0.3}, Blue, double distance=2]
&
|[alias=Fa]| Fa
\ar[d, Red, "{f:\mathrm{Hom}_{\mathcal{C}}(Fa, Gb)}"']

\\

|[alias=b]| b
\ar[r, "G"{pos=0.3}, Green, double distance=2]
&
|[alias=Gb]| Gb

\end{tikzcd}
\end{document}
```

# Морфизм категории запятой
Квадрат справа должен коммутировать.
```tikz
\usepackage{tikz}
\usepackage{tikz-cd}
\usepackage{xcolor}
\usetikzlibrary{backgrounds, fit}
\usetikzlibrary{decorations.pathmorphing}
\usetikzlibrary{shapes.geometric}
\tikzcdset{scale cd/.style=
  {every
    label/.append style={scale=#1, transform shape},
    cells={nodes={scale=#1, transform shape}},
    arrows={ultra thick, -{Stealth[scale=1]}},
  },
}
\begin{document}

\boldmath

\begin{tikzcd}[
  scale cd=2,
  sep=10em,
  /tikz/execute at end picture={
	\begin{scope}[on background layer]
		\node[
		  draw=black!80,       % Обводка (как в filldraw)
		  fill=Blue!10,        % Заливка (как в filldraw)
		  very thick,          % Толщина линии
		  ellipse,              % Форма		  
		  fit=(a) (aa),
		  label={[font=\huge]170:$\mathcal{A}$}      % Подпись СНАРУЖИ сверху
		] (caa) {};
		
		\node[
		  draw=black!80,       % Обводка (как в filldraw)
		  fill=Green!10,        % Заливка (как в filldraw)
		  very thick,          % Толщина линии
		  ellipse,              % Форма		  
		  fit=(b) (bb),
		  label={[font=\huge]170:$\mathcal{B}$}      % Подпись СНАРУЖИ сверху
		] (cbb) {};
		
		\node[
		  draw=black!80,       % Обводка (как в filldraw)
		  fill=Red!10,        % Заливка (как в filldraw)
		  very thick,          % Толщина линии
		  circle,              % Форма		  
		  fit=(Fa) (Gbb),
		  label={[font=\huge]140:$\mathcal{C}$}      % Подпись СНАРУЖИ сверху
		] (cbb) {};
	\end{scope}
  }
]

|[alias=a]| a
\ar[r, "h", Blue]
&
|[alias=aa]| a'
&
|[alias=Fa]| Fa
\ar[r, "Fh", Blue]
\ar[d, "f", Red]
&
|[alias=Faa]| Fa'
\ar[d, "f{\displaystyle '}", Red]
\\

|[alias=b]| b
\ar[r, "g", Green]
&
|[alias=bb]| b'
&
|[alias=Gb]| Gb
\ar[r, "Gg", Green]
&
|[alias=Gbb]| Gb'

\end{tikzcd}
\end{document}
```
