
## Произведение типов
```tikz
\usepackage{tikz}
\usepackage{tikz-cd}
\usepackage{xcolor}
\usetikzlibrary{backgrounds, fit}
\usetikzlibrary{decorations.pathmorphing}
\tikzcdset{scale cd/.style=
  {every
    label/.append style={scale=#1},
    cells={nodes={scale=#1}},
    arrows={
      line width=#1,
      -{Stealth[scale=1]}
    },
  },
}
\begin{document}

\boldmath

\begin{tikzcd}[
  scale cd=4,
  sep=huge
]

& |[alias=AB]| A \times B
\ar["{p_A}"', color=red, dl]
\ar["{p_B}", color=red, dr]& \\

|[alias=A]| A && B \\
& |[alias=C]| C
\ar["{f}"', ur, blue]
\ar["{g}", ul, blue]
\ar["{u}", uu, Green]
&

\end{tikzcd}
\end{document}
```

		\node[fit=(1) (I) (2), yshift=-1mm] (I_ellipse) {};

  }

&&&&&&&&&& |[alias=C]|\mathcal{C} \\

\\
\\
\\
\\

}
