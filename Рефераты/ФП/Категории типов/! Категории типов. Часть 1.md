## Категории значений

```scala
type Hom = (Any, Any)

extension (f: Hom)
  infix def ∘(g: Hom): Hom = 
    if g._2 == f._1
      then g._1 -> f._2
      else throw Exception()
  def ap(a: Any): Any = 
    if f._1 == a
      then f._2
      else throw Exception()

val id: Any => Hom =
  a => a -> a
```

```scala
val f1: Hom = (2, 4)
val f2: Hom = (4, 42)
val f3: Hom = f2 ∘ f1 ∘ id(2)

f3.ap(2) // 42
f3.ap(3) // Exception
```
