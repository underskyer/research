\begin{eqnarray}

–õ—é–±–æ–π —Ñ—É–Ω–∫—Ç–æ—Ä, –∏—Å—Ö–æ–¥—è—â–∏–π –∏–∑ $1$ –º–æ–¥–µ–ª–∏—Ä—É–µ—Ç –µ—ë –≤ –∫–æ–Ω–µ—á–Ω–æ–π –∫–∞—Ç–µ–≥–æ—Ä–∏–∏ $D$, —Ç–æ –µ—Å—Ç—å, –ø–æ —Å—É—Ç–∏, –≤—ã–±–∏—Ä–∞–µ—Ç –∏–∑ –Ω–µ—ë –µ–¥–∏–Ω—Å—Ç–≤–µ–Ω–Ω—ã–π –æ–±—ä–µ–∫—Ç. –ö–∞—Ç–µ–≥–æ—Ä–∏—è —Ç–∞–∫–∏—Ö —Ñ—É–Ω–∫—Ç–æ—Ä–æ–≤ $D^1$ –∏–∑–æ–º–æ—Ä—Ñ–Ω–∞ –¥–∏—Å–∫—Ä–µ—Ç–Ω–æ–π –ø–æ–¥–∫–∞—Ç–µ–≥–æ—Ä–∏–∏, –ø–æ—Å—Ç—Ä–æ–µ–Ω–Ω–æ–π –Ω–∞ —Ç–µ—Ö –∂–µ –æ–±—ä–µ–∫—Ç–∞—Ö —á—Ç–æ –∏ $D$. –ü–æ—ç—Ç–æ–º—É —á–∞—Å—Ç–æ –¥–ª—è –ø—Ä–æ—Å—Ç–æ—Ç—ã —ç—Ç–∏ —Ñ—É–Ω–∫—Ç–æ—Ä—ã –æ–±–æ–∑–Ω–∞—á–∞—é—Ç —Ç–∞–∫–∂–µ, –∫–∞–∫ –∏ –æ–±—ä–µ–∫—Ç—ã $d: 1 \rightarrow \mathcal{D}$.

–ü–æ–ª—É—á–∞–µ—Ç—Å—è —Å–ª–µ–¥—É—é—â–∞—è –¥–∏–∞–≥—Ä–∞–º–º–∞ –≤ –∫–∞—Ç–µ–≥–æ—Ä–∏–∏ –∫–∞—Ç–µ–≥–æ—Ä–∏–π:

## –°–æ–ø—Ä—è–∂–µ–Ω–∏—è
–†–∞–Ω–µ–µ —É–∂–µ –±—ã–ª–∏ –Ω–∞–º—ë–∫–∏, –Ω–æ —Ç–µ–ø–µ—Ä—å —Å–∫–∞–∂–µ–º –æ—Ç–∫—Ä—ã—Ç–æ: –µ—Å–ª–∏ –¥–ª—è —Ñ—É–Ω–∫—Ç–æ—Ä–∞ $F:\, \mathcal{C}\rightarrow\mathcal{D}$ —Å—É—â–µ—Å—Ç–≤—É—é—Ç –ª–µ–≤–æ–µ/–ø—Ä–∞–≤–æ–µ —Å–æ–ø—Ä—è–∂–µ–Ω–∏–µ, —Ç–æ —ç—Ç–æ –±—É–¥–µ—Ç (—Å —Ç–æ—á–Ω–æ—Å—Ç—å—é –¥–æ –∏–∑–æ–º–æ—Ä—Ñ–∏–∑–º–∞) –ø—Ä–∞–≤–æ–µ/–ª–µ–≤–æ–µ —Ä–∞—Å—à–∏—Ä–µ–Ω–∏–µ –ö–∞–Ω–∞ —Ç–æ–∂–¥–µ—Å—Ç–≤–µ–Ω–Ω–æ–≥–æ —Ñ—É–Ω–∫—Ç–æ—Ä–∞ $Id:\mathcal{C}\rightarrow\mathcal{C}$ –≤–¥–æ–ª—å $F$:
$$
\begin{eqnarray}
{Id} / F \dashv F \dashv F \backslash {Id}.
\end{eqnarray}
$$
–í [–ø—Ä–µ–¥—ã–¥—É—â–µ–π —á–∞—Å—Ç–∏ –æ–±–∑–æ—Ä–∞](https://habr.com/ru/articles/970762/#triple) –±—ã–ª–æ –ø–æ–∫–∞–∑–∞–Ω–æ, –∫–∞–∫ —Å–æ–ø—Ä—è–∂—ë–Ω–Ω—ã–µ —Ç—Ä–æ–π–∫–∏ –ø–æ—Ä–æ–∂–¥–∞—é—Ç –º–æ–Ω–∞–¥—ã –∏ –∫–æ–º–æ–Ω–∞–¥—ã. –ü—Ä–µ–¥—Å—Ç–∞–≤–ª–µ–Ω–Ω–∞—è —Ç—Ä–æ–π–∫–∞ –∏–Ω—Ç–µ—Ä–µ—Å–Ω–∞ —Ç–µ–º, —á—Ç–æ –æ–Ω–∞ –ø–∞—Ä–∞–º–µ—Ç—Ä–∏–∑–∏—Ä–æ–≤–∞–Ω–∞ –ø–æ —Å—É—Ç–∏ –µ–¥–∏–Ω—Å—Ç–≤–µ–Ω–Ω—ã–º —Ñ—É–Ω–∫—Ç–æ—Ä–æ–º $F$ ‚Äî –æ–Ω –æ–¥–∏–Ω –ø–æ—Ä–æ–∂–¥–∞–µ—Ç —Ü–µ–ª—É—é —Ä–æ—Å—Å—ã–ø—å (–∫–æ)–º–æ–Ω–∞–¥! –í —á–∞—Å—Ç–Ω–æ—Å—Ç–∏, –¥–ª—è —ç–Ω–¥–æ—Ñ—É–Ω–∫—Ç–æ—Ä–æ–≤ $F \circ (Id/F):\;\mathcal{D}\rightarrow\mathcal{D}$ –∏ $(F \backslash Id) \circ F:\;\mathcal{C}\rightarrow\mathcal{C}$ –∏–∑ –¥–∞–Ω–Ω–æ–π —Ç—Ä–æ–π–∫–∏ –º—ã –ø–æ–ª—É—á–∞–µ–º –º–æ–Ω–∞–¥—ã, –∞ –¥–ª—è $(Id/F) \circ F:\;\mathcal{C}\rightarrow\mathcal{C}$ –∏ $F \circ (F \backslash Id):\; \mathcal{D}\rightarrow\mathcal{D}$ ‚Äî –∫–æ–º–æ–Ω–∞–¥—ã. 

## –ú–æ–Ω–∞–¥–∞ –∫–æ–ø–ª–æ—Ç–Ω–æ—Å—Ç–∏

–ö–æ–Ω–µ—á–Ω–æ –∂–µ, –Ω–µ –¥–ª—è –∫–∞–∂–¥–æ–≥–æ —Ñ—É–Ω–∫—Ç–æ—Ä–∞ –º–æ–≥—É—Ç —Å—É—â–µ—Å—Ç–≤–æ–≤–∞—Ç—å –ª–µ–≤–æ–µ –∏ –ø—Ä–∞–≤–æ–µ —Å–æ–ø—Ä—è–∂–µ–Ω–∏–µ. –í —Ç–∞–∫–∏—Ö —Å–ª—É—á–∞—è—Ö —Ä–∞—Å—à–∏—Ä–µ–Ω–∏—è –ö–∞–Ω–∞ –º–æ–∂–Ω–æ —Ä–∞—Å—Å–º–∞—Ç—Ä–∏–≤–∞—Ç—å –∫–∞–∫ ¬´–Ω–∞–∏–ª—É—á—à–∏–µ –ø–æ–ø—ã—Ç–∫–∏¬ª –ø–æ—Å—Ç—Ä–æ–∏—Ç—å —Å–æ–ø—Ä—è–∂–µ–Ω–∏—è. –ï—Å–ª–∏ –¥–ª—è —Ñ—É–Ω–∫—Ç–æ—Ä–∞ $F:\, \mathcal{C}\rightarrow\mathcal{D}$ —Å–æ–ø—Ä—è–∂–µ–Ω–∏—è –≤—Å—ë –∂–µ —Å—É—â–µ—Å—Ç–≤—É—é—Ç, —Ç–æ —Å–ø—Ä–∞–≤–µ–¥–ª–∏–≤—ã–º–∏ –æ–∫–∞–∑—ã–≤–∞—é—Ç—Å—è —Å–ª–µ–¥—É—é—â–∏–µ –∏–∑–æ–º–æ—Ä—Ñ–∏–∑–º—ã:
$$
\begin{eqnarray}
F \circ (Id/F) \cong F/F,\\
F \circ (F \backslash Id) \cong F \backslash F.
\end{eqnarray}
$$
–°–æ–æ—Ç–Ω–æ—à–µ–Ω–∏—è –∫–∞–∂—É—Ç—Å—è –æ—á–µ–≤–∏–¥–Ω—ã–º–∏ —Å–æ —à–∫–æ–ª—ã, –Ω–æ –≤–≤–∏–¥—É –Ω–µ —Å–æ–≤—Å–µ–º –æ–±—ã—á–Ω—ã—Ö –æ–ø–µ—Ä–∞—Ü–∏–π ¬´–¥–µ–ª–µ–Ω–∏—è —Ñ—É–Ω–∫—Ç–æ—Ä–æ–≤¬ª, –¥–ª—è –∏—Ö –≤—ã–ø–æ–ª–Ω–µ–Ω–∏—è —Ç—Ä–µ–±—É—é—Ç—Å—è –æ–ø—Ä–µ–¥–µ–ª—ë–Ω–Ω—ã–µ —É—Å–ª–æ–≤–∏—è, –≤—Ä–æ–¥–µ –Ω–∞–ª–∏—á–∏—è —Å–æ–ø—Ä—è–∂–µ–Ω–∏—è –∏ —Å—Ç—Ä–æ–≥–æ–≥–æ –ø–æ—Ä—è–¥–∫–∞ —Ñ—É–Ω–∫—Ç–æ—Ä–æ–≤.

–ö –ø—Ä–∏–º–µ—Ä—É, —Ä–∞—Å—Å–º–æ—Ç—Ä–∏–º –ø–µ—Ä–≤—ã–π –∏–∑–æ–º–æ—Ä—Ñ–∏–∑–º, –ø—Ä–µ–¥—Å—Ç–∞–≤–ª–µ–Ω–Ω—ã–π –¥–≤—É–º—è –≤—Å—Ç—Ä–µ—á–Ω—ã–º–∏ –µ—Å—Ç–µ—Å—Ç–≤–µ–Ω–Ω—ã–º–∏ –ø—Ä–µ–æ–±—Ä–∞–∑–æ–≤–∞–Ω–∏—è–º–∏:

\begin{eqnarray}
\varphi:\;& F \circ (Id / F) &\rightsquigarrow&& F/F,\\
\psi:\;& F/F &\rightsquigarrow&& F \circ (Id / F).\\
\end{eqnarray}
$$
–û–Ω–∏ –¥–æ–ª–∂–Ω—ã —Ñ–∏–∫—Å–∏—Ä–æ–≤–∞—Ç—å—Å—è –∏–∑–≤–µ—Å—Ç–Ω—ã–º–∏ –ø—Ä–µ–æ–±—Ä–∞–∑–æ–≤–∞–Ω–∏—è–º–∏ —Å–æ–ø—Ä—è–∂–µ–Ω–∏—è, –∞ —Ç–∞–∫–∂–µ –∫–æ–µ–¥–∏–Ω–∏—Ü–µ–π —Ä–∞—Å—à–∏—Ä–µ–Ω–∏—è –ö–∞–Ω–∞ $\varepsilon_{\mathrm{Ran}}$:
$$
\begin{eqnarray}
\varepsilon_\dashv:\;& (Id / F) \circ F &\rightsquigarrow Id, \\
\eta_\dashv:\;& Id &\rightsquigarrow F \circ (Id / F), \\
\varepsilon_{\mathrm{Ran}}:\;& (F / F) \circ F &\rightsquigarrow F.
\end{eqnarray}

–° –∏—Ö –ø–æ–º–æ—â—å—é –æ–¥–Ω–∞ –≤–µ—Ç–≤—å –∏–∑–æ–º–æ—Ä—Ñ–∏–∑–º–∞ –≤—ã—Ä–∞–∂–∞–µ—Ç—Å—è —Ç–∞–∫:

\psi = (\varepsilon_{\mathrm{Ran}} \circ Id_{Id/F}) \cdot (Id_{F/F} \circ \eta_\dashv).

–ê –≤–æ—Ç –¥–ª—è —è–≤–Ω–æ–≥–æ –≤—ã—Ä–∞–∂–µ–Ω–∏—è $\varphi$ –¥–∞–Ω–Ω—ã—Ö –ø—Ä–µ–æ–±—Ä–∞–∑–æ–≤–∞–Ω–∏–π –Ω–µ –¥–æ—Å—Ç–∞—Ç–æ—á–Ω–æ. –¢—É—Ç –Ω–∞—Å –≤—ã—Ä—É—á–∏—Ç —É–Ω–∏–≤–µ—Ä—Å–∞–ª—å–Ω—ã–π –º–æ—Ä—Ñ–∏–∑–º —Ä–∞—Å—à–∏—Ä–µ–Ω–∏—è –ö–∞–Ω–∞ $u_G:\; G \rightsquigarrow F/F$, *–≤–∑–∞–∏–º–Ω–æ –æ–¥–Ω–æ–∑–Ω–∞—á–Ω–æ* —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤—É—é—â–µ–º—É –ø—Ä–µ–æ–±—Ä–∞–∑–æ–≤–∞–Ω–∏—é $\alpha_G: F \circ G \rightsquigarrow F$, –∑–∞–¥–∞–Ω–Ω–æ–≥–æ –¥–ª—è –ø—Ä–æ–∏–∑–≤–æ–ª—å–Ω–æ–≥–æ –∫–∞–Ω–¥–∏–¥–∞—Ç–∞ $G: \mathcal{D} \rightarrow \mathcal{D}$. –í—ã–±–µ—Ä–µ–º –≤ –∫–∞—á–µ—Å—Ç–≤–µ –∫–∞–Ω–¥–∏–¥–∞—Ç–∞ —Ç–æ–∂–¥–µ—Å—Ç–≤–µ–Ω–Ω—ã–π —Ñ—É–Ω–∫—Ç–æ—Ä $Id$ –≤ –∫–∞—Ç–µ–≥–æ—Ä–∏–∏ $\mathcal{D}$ —Å —Ç–æ–∂–¥–µ—Å—Ç–≤–µ–Ω–Ω—ã–º –ø—Ä–µ–æ–±—Ä–∞–∑–æ–≤–∞–Ω–∏–µ–º $\alpha_{Id}=Id_F: F \rightsquigarrow F$. –î–ª—è —Ç–∞–∫–æ–≥–æ –∫–∞–Ω–¥–∏–¥–∞—Ç–∞ —É –Ω–∞—Å –±—É–¥–µ—Ç —É–Ω–∏–≤–µ—Ä—Å–∞–ª—å–Ω—ã–π –º–æ—Ä—Ñ–∏–∑–º –≤–∏–¥–∞
$$

## –ú–æ–Ω–∞–¥–∞ –ø–µ—Ä–µ–¥–∞—á–∏ –ø—Ä–æ–¥–æ–ª–∂–µ–Ω–∏–π
